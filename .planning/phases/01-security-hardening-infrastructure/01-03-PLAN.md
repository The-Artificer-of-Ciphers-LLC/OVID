<objective>
Replace JWT-in-URL auth delivery with OAuth2 authorization code exchange, implement HttpOnly cookie token delivery, add RFC 8628 Device Authorization Flow for CLI/ARM clients, implement refresh token rotation with Redis-backed blacklist, and update the web client auth plumbing to use cookies instead of localStorage.

Purpose: This is the core security fix for Phase 1. The current JWT-in-URL pattern leaks tokens in browser history, server logs, and Referer headers. The auth code exchange eliminates this attack surface entirely.

Output: Rewritten auth flow (code exchange + cookies), device flow endpoints, refresh token rotation, updated web client auth, and comprehensive test suites.
</objective>

1. Create `api/app/auth/device_flow.py` (per D-04, RFC 8628):
   ```python
   """OAuth 2.0 Device Authorization Grant (RFC 8628) endpoints."""
   import json
   import os
   import secrets
   import string
   import time
   import uuid
   from datetime import datetime, timedelta, timezone

   from fastapi import APIRouter, Body, Depends, HTTPException, Request
   from fastapi.responses import JSONResponse

   from app.auth.deps import get_current_user
   from app.auth.jwt import create_access_token, create_refresh_token
   from app.redis import get_redis
   from app.rate_limit import limiter

   device_router = APIRouter(prefix="/v1/auth/device", tags=["device-auth"])

   def _generate_user_code(length: int = 8) -> str:
       """Generate easy-to-type alphanumeric code (no ambiguous chars)."""
       alphabet = "ABCDEFGHJKLMNPQRSTUVWXYZ23456789"  # no 0/O/1/I
       return "".join(secrets.choice(alphabet) for _ in range(length))

   @device_router.post("/authorize")
   @limiter.limit("10/minute")
   async def device_authorize(request: Request):
       redis = get_redis()
       if not redis:
           raise HTTPException(503, detail={"error": "service_unavailable"})
       device_code = secrets.token_urlsafe(32)
       user_code = _generate_user_code()
       redis.setex(f"device:{device_code}", 900, json.dumps({
           "user_code": user_code,
           "status": "pending",
           "last_poll": 0,
       }))
       redis.setex(f"usercode:{user_code}", 900, device_code)
       web_url = os.environ.get("OVID_WEB_URL", "http://localhost:3000")
       return {
           "device_code": device_code,
           "user_code": user_code,
           "verification_uri": f"{web_url}/device",
           "expires_in": 900,
           "interval": 5,
       }

   @device_router.post("/token")
   @limiter.limit("12/minute")
   async def device_token(request: Request, device_code: str = Body(..., embed=True)):
       redis = get_redis()
       if not redis:
           raise HTTPException(503, detail={"error": "service_unavailable"})
       key = f"device:{device_code}"
       raw = redis.get(key)
       if not raw:
           raise HTTPException(401, detail={"error": "expired_token"})
       data = json.loads(raw)
       # Enforce polling interval (per Pitfall 5)
       now = time.time()
       if now - data.get("last_poll", 0) < 4.5:  # slight tolerance
           raise HTTPException(429, detail={"error": "slow_down", "interval": 5})
       data["last_poll"] = now
       redis.setex(key, redis.ttl(key) or 900, json.dumps(data))
       if data["status"] == "pending":
           raise HTTPException(428, detail={"error": "authorization_pending"})
       if data["status"] == "denied":
           redis.delete(key)
           raise HTTPException(401, detail={"error": "access_denied"})
       # status == "approved"
       user_id = uuid.UUID(data["user_id"])
       redis.delete(key)
       return {
           "access_token": create_access_token(user_id),
           "refresh_token": create_refresh_token(user_id),
           "token_type": "Bearer",
           "expires_in": 3600,
       }

   @device_router.post("/approve")
   @limiter.limit("10/minute")
   async def device_approve(
       request: Request,
       user_code: str = Body(..., embed=True),
       user = Depends(get_current_user),
   ):
       redis = get_redis()
       if not redis:
           raise HTTPException(503, detail={"error": "service_unavailable"})
       uc_key = f"usercode:{user_code.upper().strip()}"
       device_code = redis.get(uc_key)
       if not device_code:
           raise HTTPException(404, detail={"error": "invalid_user_code"})
       device_code = device_code.decode()
       dc_key = f"device:{device_code}"
       raw = redis.get(dc_key)
       if not raw:
           raise HTTPException(404, detail={"error": "expired_device_code"})
       data = json.loads(raw)
       data["status"] = "approved"
       data["user_id"] = str(user.id)
       redis.setex(dc_key, redis.ttl(dc_key) or 900, json.dumps(data))
       redis.delete(uc_key)
       return {"approved": True}
   ```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| browser -> API (cookie) | HttpOnly auth cookie sent automatically on same-origin requests |
| OAuth provider -> API callback | External redirect carries auth code (not JWT) |
| CLI -> API (device flow) | CLI polls for token after user approves in browser |
| refresh token -> new tokens | Token rotation with blacklist prevents replay |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
